# Write your own code
names = ["data_quiz/wdi_co2_2005.csv", "data_quiz/wdi_co2_2010.csv", "data_quiz/wdi_co2_2015.csv"]
index = 0

for name in names:
    year = name[-8: -4]
    dataset = pd.read_csv(name)
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]
    total = co2_gasfuel_kt + co2_liquidfuel_kt 
    plt.hist(total)
    plt.title(f"Histogram for Total CO2 Emissions from Gas And Liquid Fuel (Year: {year})")
    plt.xlabel("Value")
    plt.ylabel("Frequency")
    plt.show()
    index = index + 1
